var links = document.querySelectorAll('.menu__link');

for(var i = 0; i < links.length; i++) {
	if(links[i].getAttribute('href') == window.location.pathname) {
		var link = links[i];
		
		/* добавить html-класс для стилизации */
		link.classList.add('menu__link_active');
		
		/* удалить атрибут href */
		link.removeAttribute('href');
	}
}

stylus 
/*
	Изменение велечены текста относительно viewport экрана, в приделах заданных размеров.
	Используется формула Mike Riethmuller: 
	font-size: calc([minimum size] + ([maximum size] - [minimum size]) * ((100vw - [minimum viewport width]) / ([maximum viewport width] - [minimum viewport width])));
*/

$setResponsiveFont($smallFont, $bigFont, $smallViewport, $bigViewport)
	font-size "calc(%s%s + (%s - %s) * ((100vw - %s%s) / (%s - %s)))" \
	% ($smallFont px $bigFont $smallFont $smallViewport px $bigViewport $smallViewport)
